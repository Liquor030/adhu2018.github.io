```python
# -*- coding: utf-8 -*-
from urllib.request import quote
from urllib.request import unquote

_utf8 = quote("地")
print(_utf8)
# "%E5%9C%B0"
# _gbk = quote("地","gbk")  # 也还是utf8，为什么？？？
# print(_gbk)
# "%E5%9C%B0"

utf8_ = unquote("%E5%9C%B0")
print(utf8_)
# "地"
gbk_ = unquote("%B5%D8","gbk")  # 也是这样，正着不行，反过来就行了，为什么？？？
print(gbk_)
# "地"
```
